                async fetchPrices() {
                    try {
                        // 读取本地 JSON 文件（与顶部指数同一数据源）
                        const response = await fetch('data/ca_crypto.json?t=' + Date.now());
                        const data = await response.json();
                        
                        if (data.crypto) {
                            const cryptoData = data.crypto;
                            
                            // 更新持仓价格
                            this.stocks.forEach(stock => {
                                const coinData = cryptoData[stock.code];
                                if (coinData) {
                                    stock.price = coinData.value;
                                    stock.dayChangePercent = coinData.changePercent.toFixed(2);
                                    stock.dayChange = coinData.change;
                                    stock.profitRate = ((stock.price - stock.cost) / stock.cost * 100).toFixed(2);
                                }
                            });
                        }
                        
                        this.apiStatus = 'Yahoo Finance ✓';
                        this.apiStatusClass = 'success';
                        
                        this.$nextTick(() => {
                            this.renderGauges();
                        });
                        
                    } catch (e) {
                        console.error('获取价格失败:', e);
                        this.apiStatus = '获取失败 - ' + e.message;
                        this.apiStatusClass = 'error';
                    }
                },
