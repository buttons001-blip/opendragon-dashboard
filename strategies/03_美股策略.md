### 代码实现
```javascript
async fetchPrices() {
    try {
        // 使用腾讯财经 API 获取美股实时价格和指数（国内可用）
        const symbols = 'usNVDA,usMSFT,usAMZN,usSPX,usIXIC,usDJI';
        const url = `https://qt.gtimg.cn/q=${symbols}`;
        
        const response = await fetch(url);
        const text = await response.text();
        
        // 解析个股数据
        const priceMap = {
            'NVDA': this.extractPrice(text, 'v_usNVDA'),
            'MSFT': this.extractPrice(text, 'v_usMSFT'),
            'AMZN': this.extractPrice(text, 'v_usAMZN')
        };
        
        this.stocks.forEach(stock => {
            const data = priceMap[stock.code];
            if (data) {
                stock.price = data.price;
                stock.dayChange = data.change;
                stock.dayChangePercent = data.changePercent;
                stock.profitRate = ((stock.price - stock.cost) / stock.cost * 100).toFixed(2);
            }
        });
        
        // 解析指数数据
        const spxData = this.extractPrice(text, 'v_usSPX');
        const ixicData = this.extractPrice(text, 'v_usIXIC');
        const djiData = this.extractPrice(text, 'v_usDJI');
        
        // 更新指数显示...
        
    } catch (e) {
        console.error('获取美股价格失败:', e);
    }
}

// 辅助函数：提取价格数据
extractPrice(text, varName) {
    const regex = new RegExp(`${varName}="([^"]+)"`);
    const match = text.match(regex);
    if (match && match[1]) {
        const parts = match[1].split('~');
        const price = parseFloat(parts[3]);
        const prevClose = parseFloat(parts[4]);
        const change = price - prevClose;
        const changePercent = ((change / prevClose) * 100).toFixed(2);
        return { price, change, changePercent };
    }
    return null;
}
```

## ⏰ 交易时间判断

```javascript
isTradingTime() {
    const now = new Date();
    const day = now.getDay();
    if (day === 0 || day === 6) return false; // 周末休市
    
    const time = now.getHours() * 100 + now.getMinutes();
    // 美股交易时间：21:30-04:00 (北京时间)
    return (time >= 2130 || time <= 400);
}
```
