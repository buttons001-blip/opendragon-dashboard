### 代码实现
```javascript
async fetchPrices() {
    try {
        // 从本地 JSON 文件读取加股和数字货币数据
        const response = await fetch('data/ca_crypto.json?t=' + Date.now());
        const data = await response.json();
        
        // 更新个股价格
        this.stocks.forEach(stock => {
            const stockData = data.stocks?.[stock.code];
            if (stockData) {
                stock.price = stockData.price;
                stock.dayChange = stockData.change;
                stock.dayChangePercent = stockData.changePercent;
                stock.profitRate = ((stock.price - stock.cost) / stock.cost * 100).toFixed(2);
            }
        });
        
    } catch (e) {
        console.error('获取加股价格失败:', e);
    }
}
```

## ⏰ 交易时间判断

```javascript
// 加拿大交易时间：22:30-05:00 (北京时间)
// 夏令时：21:30-04:00
isTradingTime() {
    const now = new Date();
    const day = now.getDay();
    if (day === 0 || day === 6) return false;
    
    const time = now.getHours() * 100 + now.getMinutes();
    return (time >= 2230 || time <= 500);
}
```
