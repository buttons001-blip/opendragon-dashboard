```javascript

// [0] 市场代码
// [1] 指数名称
// [2] 代码

// [4] 昨收点
```

```javascript

// 构建股票代码
const codes = this.stocks.map(s => {
    const prefix = s.code.startsWith('6') ? 'sh' : 'sz';
    return prefix + s.code;
}).join(',');

// JSONP 请求
const script = document.createElement('script');
script.src = `https://qt.gtimg.cn/q=${codes}`;
script.onload = () => this.parseTencentData();
```

## ⏰ 交易时间判断

```javascript
isTradingTime() {
    const now = new Date();
    const day = now.getDay();
    if (day === 0 || day === 6) return false; // 周末休市
    
    const time = now.getHours() * 100 + now.getMinutes();
    // 上午 9:30-11:30, 下午 13:00-15:00
    return (time >= 930 && time <= 1130) || (time >= 1300 && time <= 1500);
}
```
