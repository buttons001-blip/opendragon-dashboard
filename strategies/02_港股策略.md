### API 响应格式
```javascript
v_hk01810="1~小米集团~01810~34.88~34.50~..."
// 字段说明：
// [3] 当前价
// [4] 昨收价
```

### 代码实现
```javascript
async fetchPrices() {
    try {
        // 港股使用腾讯 API，前缀为 hk
        const codes = this.stocks.map(s => 'hk' + s.code).join(',');
        const url = `https://qt.gtimg.cn/q=${codes}`;
        
        const response = await fetch(url);
        const text = await response.text();
        
        // 解析腾讯 API 返回的数据
        this.stocks.forEach(stock => {
            const varName = `v_hk${stock.code}`;
            const regex = new RegExp(`${varName}="([^"]+)"`);
            const match = text.match(regex);
            
            if (match && match[1]) {
                const parts = match[1].split('~');
                const currentPrice = parseFloat(parts[3]);
                const prevClose = parseFloat(parts[4]);
                
                stock.price = currentPrice;
                stock.prevClose = prevClose;
                stock.dayChange = currentPrice - prevClose;
                stock.dayChangePercent = ((stock.dayChange / prevClose) * 100).toFixed(2);
                stock.profitRate = ((currentPrice - stock.cost) / stock.cost * 100).toFixed(2);
            }
        });
    } catch (e) {
        console.error('获取港股价格失败:', e);
    }
}
```

## ⏰ 交易时间判断

```javascript
isTradingTime() {
    const now = new Date();
    const day = now.getDay();
    if (day === 0 || day === 6) return false; // 周末休市
    
    const time = now.getHours() * 100 + now.getMinutes();
    // 港股交易时间：9:30-12:00, 13:00-16:00 (北京时间)
    return (time >= 930 && time <= 1200) || (time >= 1300 && time <= 1600);
}
```
